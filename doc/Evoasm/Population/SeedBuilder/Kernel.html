def method_missing(name, *args, **kwargs, &block)
  if builder.instruction_name? name
    if builder.allowed_instruction_name? name
      @instructions << [name, kwargs]
    else
      raise ArgumentError, "'#{name}' is not in the current instruction set"
    end
  else
    super
  end
end
